#version 430

layout (triangles) in;
layout (triangle_strip, max_vertices = 3) out;

in vec2 pass_textureCoordinates[];
in vec3 pass_surfaceNormal[];
in vec3 pass_toLightVector[][5];

out vec2 textureCoordinates;
out vec3 surfaceNormal;
out vec3 toLightVector[5];

void main(void)
{

	textureCoordinates = pass_textureCoordinates[0];
	surfaceNormal = pass_surfaceNormal[0];
	toLightVector = pass_toLightVector[0];
	
	EmitVertex();
	
	textureCoordinates = pass_textureCoordinates[1];
	surfaceNormal = pass_surfaceNormal[1];
	toLightVector = pass_toLightVector[1];
	
	EmitVertex();
	
	textureCoordinates = pass_textureCoordinates[2];
	surfaceNormal = pass_surfaceNormal[2];
	toLightVector = pass_toLightVector[2];
	
	EmitVertex();

	EndPrimitive();

}
